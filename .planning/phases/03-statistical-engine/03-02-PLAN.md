**packages/analysis/schemas/requests.py:**

```python
from pydantic import BaseModel
from datetime import date
from typing import Optional

class MetricRow(BaseModel):
    date: date
    spend_usd: float
    revenue: float
    conversions: float = 0.0

class HolidayEvent(BaseModel):
    name: str
    date: date
    lower_window: int = 0
    upper_window: int = 0

class ForecastRequest(BaseModel):
    tenant_id: str
    campaign_id: str
    metrics: list[MetricRow]
    user_events: list[HolidayEvent] = []
    forecast_days: int = 90

class IncrementalityRequest(BaseModel):
    tenant_id: str
    campaign_id: str
    metrics: list[MetricRow]
    intervention_date: date  # budget change or campaign start date
    confidence_level: float = 0.94  # HDI probability

class SaturationRequest(BaseModel):
    tenant_id: str
    campaign_id: str
    metrics: list[MetricRow]  # needs spend variation to fit curve
    recent_days: int = 30  # window for current spend average

class AnomalyRequest(BaseModel):
    tenant_id: str
    campaign_id: str
    metrics: list[MetricRow]
    threshold_sigma: float = 2.5  # standard deviations for anomaly detection
```

**packages/analysis/schemas/responses.py:**

```python
from pydantic import BaseModel
from datetime import date
from typing import Optional

class ForecastPoint(BaseModel):
    date: date
    yhat: float
    yhat_lower: float
    yhat_upper: float

class SeasonalComponent(BaseModel):
    date: date
    trend: float
    yearly: float
    weekly: float
    holidays: float = 0.0

class ForecastResponse(BaseModel):
    forecast: list[ForecastPoint]
    components: list[SeasonalComponent]
    model_params: dict = {}  # Prophet model parameters for debugging

class IncrementalityResponse(BaseModel):
    lift_mean: float
    lift_lower: float
    lift_upper: float
    confidence: float
    cumulative_lift: float  # total lift over post-period
    pre_period_mean: float
    post_period_mean: float
    counterfactual_mean: float  # what would have happened without intervention
    diagnostics: dict = {}  # R-hat, ESS for model quality checks

class SaturationResponse(BaseModel):
    saturation_percent: Optional[float]  # 0.0-1.0, None if fitting failed
    hill_alpha: Optional[float]
    hill_mu: Optional[float]
    hill_gamma: Optional[float]
    status: str  # 'estimated' | 'insufficient_variation' | 'error'
    error: Optional[str] = None

class AnomalyRecord(BaseModel):
    date: date
    actual: float
    expected: float
    deviation_sigma: float
    direction: str  # 'spike' | 'dip'

class AnomalyResponse(BaseModel):
    anomalies: list[AnomalyRecord]
    seasonal_strength: float  # 0-1 measure of how seasonal the data is
    trend_direction: str  # 'increasing' | 'decreasing' | 'stable'
```
